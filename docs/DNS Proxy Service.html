#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#include <netinet/in.h>

#define MAX_RULES 100
#define MAX_SERVERS 10
#define DNS_PORT 53
#define BUFFER_SIZE 512
#define CONFIG_FILE "dns_proxy.conf"

// –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –¥–ª—è —Ñ–∏–ª—å—Ç—Ä–∞ —Ç–∏–ø–æ–≤
typedef struct {
    int ipv4;    // A
    int ipv6;    // AAAA
    int mx;      // MX
    int ns;      // NS
    int all;     // *
} TypeFilter;

// –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –¥–ª—è –ø—Ä–∞–≤–∏–ª–∞ –º–∞—Ä—à—Ä—É—Ç–∏–∑–∞—Ü–∏–∏
typedef struct {
    char pattern[128];           // –®–∞–±–ª–æ–Ω –¥–æ–º–µ–Ω–∞
    TypeFilter filter;          // –§–∏–ª—å—Ç—Ä —Ç–∏–ø–æ–≤
    char servers[MAX_SERVERS][256]; // DNS —Å–µ—Ä–≤–µ—Ä—ã
    int server_count;           // –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–µ—Ä–≤–µ—Ä–æ–≤
} DnsRule;

// –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏
typedef struct {
    DnsRule rules[MAX_RULES];
    int rule_count;
    char default_servers[MAX_SERVERS][256];
    int default_server_count;
    int debug;  // –§–ª–∞–≥ –æ—Ç–ª–∞–¥–∫–∏
} DnsConfig;

typedef struct {
    int socket;
    struct sockaddr_in client_addr;
    socklen_t client_addr_len;
    unsigned char buffer[BUFFER_SIZE];
    size_t buffer_len;
} QueryContext;

DnsConfig g_config;

// –í—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏

void debug_log(const char *fmt, ...) {
    if (!g_config.debug) return;
    
    va_list args;
    va_start(args, fmt);
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    char timestamp[32];
    strftime(timestamp, sizeof(timestamp), "%Y-%m-%d %H:%M:%S", tm_info);
    
    printf("[%s] DEBUG: ", timestamp);
    vprintf(fmt, args);
    va_end(args);
}

// –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –¥–æ–º–µ–Ω–Ω–æ–≥–æ –∏–º–µ–Ω–∏ —à–∞–±–ª–æ–Ω—É
int domain_matches(const char *domain, const char *pattern) {
    char domain_lower[256];
    char pattern_lower[256];
    
    // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ –Ω–∏–∂–Ω–∏–π —Ä–µ–≥–∏—Å—Ç—Ä
    for (int i = 0; domain[i]; i++) {
        domain_lower[i] = tolower(domain[i]);
    }
    domain_lower[strlen(domain)] = '\0';
    
    for (int i = 0; pattern[i]; i++) {
        pattern_lower[i] = tolower(pattern[i]);
    }
    pattern_lower[strlen(pattern)] = '\0';
    
    // –û–±—Ä–∞–±–æ—Ç–∫–∞ wildcards
    if (strchr(pattern_lower, '*') == NULL) {
        return strcmp(domain_lower, pattern_lower) == 0;
    }
    
    char *star = strchr(pattern_lower, '*');
    int prefix_len = star - pattern_lower;
    int suffix_len = strlen(star + 1);
    
    if (strlen(domain_lower) < prefix_len + suffix_len) {
        return 0;
    }
    
    if (strncmp(domain_lower, pattern_lower, prefix_len) != 0) {
        return 0;
    }
    
    if (strcmp(domain_lower + strlen(domain_lower) - suffix_len, star + 1) != 0) {
        return 0;
    }
    
    return 1;
}

// –ü–∞—Ä—Å–∏–Ω–≥ DNS –ø–∞–∫–µ—Ç–∞ –∏ –∏–∑–≤–ª–µ—á–µ–Ω–∏–µ –¥–æ–º–µ–Ω–Ω–æ–≥–æ –∏–º–µ–Ω–∏
int extract_domain(const unsigned char *buffer, size_t len, char *domain) {
    if (len < 12) return -1;
    
    int pos = 12;
    domain[0] = '\0';
    
    while (pos < len && buffer[pos] != 0) {
        unsigned char len_byte = buffer[pos++];
        
        if (len_byte >= 192) {
            pos++;
            break;
        }
        
        if (domain[0] != '\0') strcat(domain, ".");
        
        for (int i = 0; i < len_byte && pos < len; i++, pos++) {
            char c = buffer[pos];
            strncat(domain, &c, 1);
        }
    }
    
    return 0;
}

// –†–µ–∑–æ–ª—é—Ü–∏—è –∏–º–µ–Ω–∏ DNS —Å–µ—Ä–≤–µ—Ä–∞ –≤ IP
char* resolve_server(const char *server) {
    static char resolved_ip[16];
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —ç—Ç–æ —É–∂–µ IP –∞–¥—Ä–µ—Å–æ–º
    if (inet_pton(AF_INET, server, &((struct in_addr){0})) == 1) {
        strcpy(resolved_ip, server);
        return resolved_ip;
    }
    
    // –ü–æ–ø—ã—Ç–∫–∞ —Ä–µ–∑–æ–ª—é—Ü–∏–∏ –ø–æ –∏–º–µ–Ω–∏
    struct hostent *he = gethostbyname(server);
    if (he == NULL) {
        debug_log("Failed to resolve %s\n", server);
        return NULL;
    }
    
    strcpy(resolved_ip, inet_ntoa(*((struct in_addr *)he->h_addr)));
    debug_log("Resolved %s to %s\n", server, resolved_ip);
    
    return resolved_ip;
}

// –ó–∞–≥—Ä—É–∑–∫–∞ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏ –∏–∑ —Ñ–∞–π–ª–∞
int load_config(const char *filename) {
    FILE *f = fopen(filename, "r");
    if (f == NULL) {
        perror("Cannot open config file");
        return -1;
    }
    
    char line[512];
    g_config.rule_count = 0;
    g_config.default_server_count = 0;
    
    while (fgets(line, sizeof(line), f)) {
        // –£–¥–∞–ª–µ–Ω–∏–µ –ø—Ä–æ–±–µ–ª–æ–≤ –∏ –∫–æ–º–º–µ–Ω—Ç–∞—Ä–∏–µ–≤
        line[strcspn(line, "\n")] = 0;
        if (line[0] == '#' || line[0] == '\0') continue;
        
        char *ptr = line;
        while (*ptr == ' ') ptr++;
        
        if (strncmp(ptr, "debug", 5) == 0) {
            g_config.debug = 1;
            printf("Debug mode enabled\n");
            continue;
        }
        
        if (strncmp(ptr, "default:", 8) == 0) {
            char *servers = ptr + 8;
            char *token = strtok(servers, " ");
            
            while (token && g_config.default_server_count < MAX_SERVERS) {
                strcpy(g_config.default_servers[g_config.default_server_count++], token);
                token = strtok(NULL, " ");
            }
            continue;
        }
        
        if (strncmp(ptr, "rule:", 5) == 0 && g_config.rule_count < MAX_RULES) {
            DnsRule *rule = &g_config.rules[g_config.rule_count++];
            
            char *pattern = ptr + 5;
            while (*pattern == ' ') pattern++;
            
            char *arrow1 = strstr(pattern, "->");
            char *arrow2 = strstr(arrow1 + 2, "->");
            
            int plen = arrow1 - pattern;
            strncpy(rule->pattern, pattern, plen);
            rule->pattern[plen] = '\0';
            
            // –£–¥–∞–ª–µ–Ω–∏–µ –ø—Ä–æ–±–µ–ª–æ–≤
            while (plen > 0 && rule->pattern[plen-1] == ' ') 
                rule->pattern[--plen] = '\0';
            
            // –ü–∞—Ä—Å–∏–Ω–≥ —Ñ–∏–ª—å—Ç—Ä–∞ —Ç–∏–ø–æ–≤
            char *filter_str = arrow1 + 2;
            while (*filter_str == ' ') filter_str++;
            
            if (strncmp(filter_str, "ipv4", 4) == 0) {
                rule->filter.ipv4 = 1;
            } else if (strncmp(filter_str, "ipv6", 4) == 0) {
                rule->filter.ipv6 = 1;
            } else if (strncmp(filter_str, "mx", 2) == 0) {
                rule->filter.mx = 1;
            } else if (strncmp(filter_str, "ns", 2) == 0) {
                rule->filter.ns = 1;
            } else {
                rule->filter.all = 1;
            }
            
            // –ü–∞—Ä—Å–∏–Ω–≥ —Å–µ—Ä–≤–µ—Ä–æ–≤
            char *servers = arrow2 + 2;
            while (*servers == ' ') servers++;
            
            char servers_copy[256];
            strcpy(servers_copy, servers);
            
            char *token = strtok(servers_copy, " ");
            rule->server_count = 0;
            
            while (token && rule->server_count < MAX_SERVERS) {
                strcpy(rule->servers[rule->server_count++], token);
                token = strtok(NULL, " ");
            }
        }
    }
    
    fclose(f);
    printf("Loaded %d rules, %d default servers\n", 
           g_config.rule_count, g_config.default_server_count);
    
    return 0;
}

// –û—Ç–ø—Ä–∞–≤–∫–∞ DNS –∑–∞–ø—Ä–æ—Å–∞ –Ω–∞ upstream —Å–µ—Ä–≤–µ—Ä
int forward_query(const char *server_ip, 
                        const unsigned char *query, 
                        size_t query_len,
                        unsigned char *response, 
                        size_t *response_len) {
    
    int sock = socket(AF_INET, SOCK_DGRAM, 0);
    if (sock < 0) {
        perror("socket");
        return -1;
    }
    
    struct sockaddr_in server;
    memset(&server, 0, sizeof(server));
    server.sin_family = AF_INET;
    server.sin_port = htons(53);
    inet_pton(AF_INET, server_ip, &server.sin_addr);
    
    struct timeval timeout;
    timeout.tv_sec = 3;
    timeout.tv_usec = 0;
    setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout));
    
    if (sendto(sock, query, query_len, 0, 
               (struct sockaddr *)&server, sizeof(server)) < 0) {
        perror("sendto");
        close(sock);
        return -1;
    }
    
    struct sockaddr_in from;
    socklen_t from_len = sizeof(from);
    
    ssize_t n = recvfrom(sock, response, BUFFER_SIZE, 0, 
                          (struct sockaddr *)&from, &from_len);
    
    if (n < 0) {
        perror("recvfrom");
        close(sock);
        return -1;
    }
    
    *response_len = n;
    close(sock);
    
    return 0;
}

// –û–±—Ä–∞–±–æ—Ç–∫–∞ DNS –∑–∞–ø—Ä–æ—Å–∞ –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ
void* handle_query(void *arg) {
    QueryContext *ctx = (QueryContext *)arg;
    
    char domain[256];
    if (extract_domain(ctx->buffer, ctx->buffer_len, domain) < 0) {
        debug_log("Failed to extract domain\n");
        free(ctx);
        return NULL;
    }
    
    printf("Query for: %s\n", domain);
    debug_log("Received query for domain: %s\n", domain);
    
    // –ü–æ–∏—Å–∫ –ø–æ–¥—Ö–æ–¥—è—â–µ–≥–æ –ø—Ä–∞–≤–∏–ª–∞
    DnsRule *matched_rule = NULL;
    for (int i = 0; i < g_config.rule_count; i++) {
        if (domain_matches(domain, g_config.rules[i].pattern)) {
            matched_rule = &g_config.rules[i];
            debug_log("Matched rule: %s\n", matched_rule->pattern);
            break;
        }
    }
    
    // –í—ã–±–æ—Ä —Å–µ—Ä–≤–µ—Ä–æ–≤
    int server_count = 0;
    char **servers = NULL;
    
    if (matched_rule) {
        server_count = matched_rule->server_count;
        servers = (char **)matched_rule->servers;
        debug_log("Using %d servers from rule\n", server_count);
    } else {
        server_count = g_config.default_server_count;
        servers = (char **)g_config.default_servers;
        debug_log("Using %d default servers\n", server_count);
    }
    
    // –ü–æ–ø—ã—Ç–∫–∞ –æ—Ç–ø—Ä–∞–≤–∏—Ç—å –∑–∞–ø—Ä–æ—Å –Ω–∞ –∫–∞–∂–¥—ã–π —Å–µ—Ä–≤–µ—Ä
    unsigned char response[BUFFER_SIZE];
    size_t response_len = 0;
    int success = 0;
    
    for (int i = 0; i < server_count; i++) {
        char *server_ip = resolve_server(servers[i]);
        
        if (server_ip == NULL) {
            debug_log("Cannot resolve %s, trying next\n", servers[i]);
            continue;
        }
        
        debug_log("Forwarding to %s (%s)\n", servers[i], server_ip);
        
        if (forward_query(server_ip, ctx->buffer, ctx->buffer_len, 
                            response, &response_len) == 0) {
            success = 1;
            break;
        }
    }
    
    // –û—Ç–ø—Ä–∞–≤–∫–∞ –æ—Ç–≤–µ—Ç–∞ –∫–ª–∏–µ–Ω—Ç—É
    if (success) {
        sendto(ctx->socket, response, response_len, 0,
               (struct sockaddr *)&ctx->client_addr, ctx->client_addr_len);
        debug_log("Response sent (%zu bytes)\n", response_len);
    } else {
        debug_log("All upstream servers failed\n");
    }
    
    free(ctx);
    return NULL;
}

// –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª —Å–µ—Ä–≤–µ—Ä–∞
int main(int argc, char *argv[]) {
    int sock;
    struct sockaddr_in addr, client_addr;
    socklen_t client_addr_len;
    
    // –ó–∞–≥—Ä—É–∑–∫–∞ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏
    if (load_config(CONFIG_FILE) < 0) {
        fprintf(stderr, "Using default configuration\n");
        strcpy(g_config.default_servers[0], "8.8.8.8");
        g_config.default_server_count = 1;
    }
    
    // –°–æ–∑–¥–∞–Ω–∏–µ UDP —Å–æ–∫–µ—Ç–∞
    sock = socket(AF_INET, SOCK_DGRAM, 0);
    if (sock < 0) {
        perror("socket");
        return 1;
    }
    
    int reuse = 1;
    setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse));
    
    // –ü—Ä–∏–≤—è–∑–∫–∞ –∫ –ø–æ—Ä—Ç—É 53
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = htonl(INADDR_ANY);
    addr.sin_port = htons(DNS_PORT);
    
    if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("bind");
        return 1;
    }
    
    printf("DNS Proxy listening on port %d\n", DNS_PORT);
    if (g_config.debug) {
        printf("Debug mode: ON\n");
    }
    
    // –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª
    while (1) {
        QueryContext *ctx = malloc(sizeof(QueryContext));
        ctx->socket = sock;
        ctx->client_addr_len = sizeof(client_addr);
        
        ssize_t n = recvfrom(sock, ctx->buffer, BUFFER_SIZE, 0,
                            (struct sockaddr *)&client_addr, &ctx->client_addr_len);
        
        if (n < 0) {
            perror("recvfrom");
            free(ctx);
            continue;
        }
        
        ctx->buffer_len = n;
        ctx->client_addr = client_addr;
        
        // –°–æ–∑–¥–∞–Ω–∏–µ –Ω–æ–≤–æ–≥–æ –ø–æ—Ç–æ–∫–∞ –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∑–∞–ø—Ä–æ—Å–∞
        pthread_t thread;
        pthread_create(&thread, NULL, handle_query, ctx);
        pthread_detach(thread);
    }
    
    close(sock);
    return 0;
}

// –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —É—Ç–∏–ª–∏—Ç—ã –∏ —Ñ—É–Ω–∫—Ü–∏–∏

// –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ —Å –≤—Ä–µ–º–µ–Ω–Ω–æ–π –º–µ—Ç–∫–æ–π
void log_timestamp() {
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    printf("[%04d-%02d-%02d %02d:%02d:%02d] ",
           tm_info->tm_year + 1900,
           tm_info->tm_mon + 1,
           tm_info->tm_mday,
           tm_info->tm_hour,
           tm_info->tm_min,
           tm_info->tm_sec);
}

// –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ —Ç–∏–ø–∞ –∑–∞–ø–∏—Å–∏ –≤ —Å—Ç—Ä–æ–∫—É
const char* qtype_to_string(int qtype) {
    switch (qtype) {
        case 1: return "A";
        case 2: return "NS";
        case 5: return "CNAME";
        case 15: return "MX";
        case 28: return "AAAA";
        default: return "UNKNOWN";
    }
}

// –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –æ—Ç–≤–µ—Ç–∞ –ø–æ —Ç–∏–ø–∞–º –∑–∞–ø–∏—Å–µ–π
int should_filter(TypeFilter filter, int qtype) {
    if (filter.all) return 0; // –ù–µ —Ñ–∏–ª—å—Ç—Ä–æ–≤–∞—Ç—å
    
    switch (qtype) {
        case 1: return !filter.ipv4;
        case 28: return !filter.ipv6;
        case 15: return !filter.mx;
        case 2: return !filter.ns;
        default: return 1;
    }
}

void print_config(DnsConfig *config) {
    printf("\n=== DNS Proxy Configuration ===\n");
    printf("Rules count: %d\n", config->rule_count);
    
    for (int i = 0; i < config->rule_count; i++) {
        DnsRule *rule = &config->rules[i];
        printf("\nRule %d:\n", i + 1);
        printf("  Pattern: %s\n", rule->pattern);
        printf("  Servers: ");
        for (int j = 0; j < rule->server_count; j++) {
            printf("%s ", rule->servers[j]);
        }
        printf("\n");
    }
    
    printf("\nDefault servers: ");
    for (int i = 0; i < config->default_server_count; i++) {
        printf("%s ", config->default_servers[i]);
    }
    printf("\n\n");
}
